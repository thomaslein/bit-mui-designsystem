import { Stack } from '@kreftregisteret/designsystem.layout.stack';
import DeleteIcon from '@mui/icons-material/Delete';
import { Button } from './button.js';
import { LoadingButton } from './loading-button.js';
import { IconButton } from './icon-button.js';

Buttons communicate actions that users can take. They are typically placed throughout your UI, in places like:

- Modal windows
- Forms
- Cards
- Toolbars

## Basic usage

The button component provides the standard `Button` component, the experimental `LoadingButton` component, and the `IconButton` component.

```ts
import {
  Button,
  LoadingButton,
  IconButton,
} from '@kreftregisteret/designsystem.inputs.button';
```

```js live
() => (
  <Stack spacing={2}>
    <Button variant="contained">Button</Button>
    <LoadingButton loading variant="contained">
      Loading Button
    </LoadingButton>
    <IconButton aria-label="delete">
      <DeleteIcon />
    </IconButton>
  </Stack>
);
```
