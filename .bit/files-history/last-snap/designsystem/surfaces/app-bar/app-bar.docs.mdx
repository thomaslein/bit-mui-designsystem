import MenuIcon from '@mui/icons-material/Menu';
import { Box } from '@kreftregisteret/designsystem.layout.box';
import { Typography } from '@kreftregisteret/designsystem.data-display.typography';
import { Button, IconButton } from '@kreftregisteret/designsystem.inputs.button';
import { AppBar } from './app-bar.js';
import { Toolbar } from './toolbar.js';

The top App bar provides content and actions related to the current screen. It's used for branding, screen titles, navigation, and actions.

It can transform into a contextual action bar or be used as a navbar.

## Basic usage

```ts
import { AppBar, Toolbar } from '@kreftregisteret/designsystem.surfaces.app-bar';
```

```js live
() => {
  return (
    <Box sx={{ flexGrow: 1, width: 400 }}>
      <AppBar position="static">
        <Toolbar>
          <IconButton
            size="large"
            edge="start"
            color="inherit"
            aria-label="menu"
            sx={{ mr: 2 }}
          >
            <MenuIcon />
          </IconButton>
          <Typography variant="h6" component="div" sx={{ flexGrow: 1 }}>
            News
          </Typography>
          <Button color="inherit">Login</Button>
        </Toolbar>
      </AppBar>
    </Box>
  );
};
```
